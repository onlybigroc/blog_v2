1 /\*修改原字段名bh为bh\_tmp\*/
2 alter table Tab\_Name rename column bh to bh\_tmp;
3 /\*增加一个和原字段名同名的字段bh\*/
4 alter table Tab\_Name add bh varchar2(500);
5 /\*将原字段bh\_tmp数据更新到增加的字段bh\*/
6 update Tab\_Name set bh\=trim(bh\_tmp);
7 /\*更新完，删除原字段bh\_tmp\*/
8 alter table Tab\_Name drop column bh1;
