MERGE INTO RDP\_CHARTS\_SETTING T1
USING (SELECT '10001' AS PAGE\_ID, 'test' AS CHART\_OPTION FROM DUAL) T2
ON (T1.PAGE\_ID \= T2.PAGE\_ID)
WHEN MATCHED THEN
  UPDATE SET T1.CHART\_OPTION \= T2.CHART\_OPTION
WHEN NOT MATCHED THEN
  INSERT
    (T1.PAGE\_ID, T1.CHART\_OPTION)
  VALUES
    (T2.PAGE\_ID, T2.CHART\_OPTION);

RDP\_CHARTS\_SETTING ：表  

PAGE\_ID：字段1

CHART\_OPTION：字段2

10001:数据1

test：数据2
